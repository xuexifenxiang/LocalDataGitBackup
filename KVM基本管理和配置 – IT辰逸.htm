####virsh命令交互####
[root@WDZJ-KVM[2] ~]# virsh
Welcome to virsh, the virtualization interactive terminal.
 
Type:  'help' for help with commands
       'quit' to quit
 
virsh # help
Grouped commands:
 
Domain Management (help keyword 'domain'):
    attach-device                  attach device from an XML file
    attach-disk                    attach disk device
 
####查看所有kvm的状态####
 
[root@WDZJ-KVM[2] ~]# virsh list --all
Id    Name                           State
----------------------------------------------------
-     centos6.5                      shut off
 
###磁盘状态####
[root@WDZJ-KVM[2] ~]# virsh domblklist centos6.5
Target     Source
------------------------------------------------
vda        /data/kvm-disk/centos6.5.qcow2
hdc        -
 
###网卡状态###
[root@WDZJ-KVM[2] ~]# virsh domifstat centos6.5 vnet0
vnet0 rx_bytes 1376493
vnet0 rx_packets 18133
vnet0 rx_errs 0
vnet0 rx_drop 0
vnet0 tx_bytes 609
vnet0 tx_packets 10
vnet0 tx_errs 0
vnet0 tx_drop 0
 
####网卡列表####
[root@WDZJ-KVM[2] ~]# virsh domiflist centos6.5
Interface  Type       Source     Model       MAC
-------------------------------------------------------
vnet0      bridge     wdzj_br0   virtio      52:54:00:80:0d:d4
 
###域信息###
[root@WDZJ-KVM[2] ~]# virsh dominfo centos6.5
Id:             11
Name:           centos6.5
UUID:           b131cfbf-f061-9696-05de-2fab6a1eaf1e

Security DOI:   0
 
 
####让指定kvm开机####
[root@WDZJ-KVM[2] ~]# virsh start centos6.5
Domain centos6.5 started
 
####让指定KVM关机####
##默认情况下virsh 不能直接操作linux使其关机，故此必须配置启用acpid(linux操作系统新型电源管理标准)##
[root@wdzj-centos ~]# yum install acpid
[root@wdzj-centos ~]# chkconfig acpid on
[root@wdzj-centos ~]# service acpid start
Starting acpi daemon:                                      [  OK  ]
 
####关的太快,抓图都没抓到，艾玛###
[root@WDZJ-KVM[2] ~]# virsh shutdown centos6.5
Domain centos6.5 is being shutdown
 
####状态####
[root@WDZJ-KVM[2] ~]# virsh list --all
Id    Name                           State
----------------------------------------------------
-     centos6.5                      shut off
 
###强制关闭电源（慎用）###
[root@WDZJ-KVM[2] ~]# virsh destroy centos6.5
 
####创建配置文件启动KVM###
[root@WDZJ-KVM[2] ~]# virsh create /etc/libvirt/qemu/centos6.5.xml
Domain centos6.5 created from /etc/libvirt/qemu/centos6.5.xml
 
[root@WDZJ-KVM[2] ~]# virsh list
Id    Name                           State
----------------------------------------------------
11    centos6.5                      running
 
####开机自启动KVM####
[root@WDZJ-KVM[2] ~]# virsh autostart centos6.5
Domain centos6.5 marked as autostarted
 
[root@WDZJ-KVM[2] ~]# ll /etc/libvirt/qemu/autostart/
total 0
lrwxrwxrwx 1 root root 31 Jun 21 00:21 centos6.5.xml -> /etc/libvirt/qemu/centos6.5.xml
 
###配置文件导出####
[root@WDZJ-KVM[2] ~]# virsh dumpxml centos6.5 > ./centos6.5-xml
[root@WDZJ-KVM[2] ~]# ll centos6.5-xml
-rw-r--r-- 1 root root 2604 Jun 21 00:24 centos6.5-xml
 
###删除KVM####  #这个命令只删除配置文件，并不会删除磁盘文件#
[root@WDZJ-KVM[2] ~]# virsh undefine centos6.5
Domain centos6.5 has been undefined
 
####挂起KVM###
[root@WDZJ-KVM[2] ~]# virsh suspend centos6.5
Domain centos6.5 suspended
 
####恢复挂起####
[root@WDZJ-KVM[2] ~]# virsh resume centos6.5
Domain centos6.5 resumed
 
###创建快照####
[root@WDZJ-KVM[2] ~]# virsh edit centos6.5
#####使用xml文件创建快照，添加domainsnapshot 标签即可####
[root@WDZJ-KVM[2] ~]# cat snapshot-for-centos6.5.xml
 
  snapshot-one
  Snapshot Test
  
    
    
  
 
 
###保存退出后，创建快照####
[root@WDZJ-KVM[2] ~]# virsh snapshot-create centos6.5 snapshot-for-centos6.5.xml
Domain snapshot snapshot-one created from 'snapshot-for-centos6.5.xml'
 
###查看当前快照###
[root@WDZJ-KVM[2] ~]# virsh snapshot-list centos6.5
Name                 Creation Time             State
------------------------------------------------------------
snapshot-one         2014-06-21 00:54:05 +0800 running
 
###tree 树形显示当前快照###
[root@WDZJ-KVM[2] ~]# virsh snapshot-list centos6.5 --tree
snapshot-one
 
####恢复快照####  ##如果当前虚拟机器正在运行，请关机后执行，如果不能关机，请加上 --force 参数###
[root@WDZJ-KVM[2] ~]# virsh snapshot-revert centos6.5 snapshot-one
 
####删除快照####
[root@WDZJ-KVM[2] ~]# virsh snapshot-delete centos6.5 snapshot-one
Domain snapshot snapshot-one deleted

###创建快照###
[root@WDZJ-KVM[2] ~]# qemu-img snapshot -c cnapshot-one /data/kvm-disk/centos6.5.qcow2
 
####列出快照清单###
[root@WDZJ-KVM[2] ~]# qemu-img snapshot -l /data/kvm-disk/centos6.5.qcow2
Snapshot list:
 
####使用快照####
[root@WDZJ-KVM[2] ~]# qemu-img snapshot -a cnapshot-one /data/kvm-disk/centos6.5.qcow2
 
####删除快照####
[root@WDZJ-KVM[2] ~]# qemu-img snapshot -d cnapshot-one /data/kvm-disk/centos6.5.qcow2
ID        TAG                 VM SIZE                DATE       VM CLOCK
